cls
@ECHO OFF
title Encrypted Folder
if EXIST "HTG Locker" goto UNLOCK
if NOT EXIST Encrypted Folder goto MDLOCKER
:CONFIRM
echo Are you sure you want to lock the folder? (Y/N)
set/p "cho=(more than)"
if %cho%==Y goto LOCK
if %cho%==y goto LOCK
if %cho%==n goto END
if %cho%==N goto END
echo Invalid choice
goto CONFIRM
:LOCK
ren Encrypted Folder "HTG Locker"
attrib +h +s "HTG Locker"
echo Folder Locked
goto END
:UNLOCK
echo Enter password to unlock folder
set/p "pass=(more than)"
if NOT %pass%== PASSWORD_GOES_HERE goto FAIL
attrib -h -s "HTG Locker"
ren "HTG Locker" Encrypted Folder
echo Folder Unlocked
goto end
:MDLOCKER
md Private
echo Private created successfully
goto End
:End
